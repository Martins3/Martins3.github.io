### vsyscall
- 将 vsyscall 映射到固定的位置为什么存在潜在的风险

```c
#include <stdio.h>
#include <time.h>

typedef time_t (*time_func)(time_t *);

int main(int argc, char *argv[]) {
  time_t tloc;
  int retval = 0;

  time_func func = (time_func)0xffffffffff600000;

  retval = func(&tloc);
  if (retval < 0) {
    perror("time_func");
    return -1;
  }
  printf("%ld\n", tloc);

  return 0;
}
```
